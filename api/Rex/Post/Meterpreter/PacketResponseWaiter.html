def wait(interval)
  if( interval and interval == -1 )
    while(not self.done)
      ::IO.select(nil, nil, nil, 0.1)
    end
  else
    begin
      Timeout.timeout(interval) {
        while(not self.done)
          ::IO.select(nil, nil, nil, 0.1)
        end
      }
    rescue Timeout::Error
      self.response = nil
